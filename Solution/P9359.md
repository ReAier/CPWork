## [$P9359$](https://www.luogu.com.cn/problem/P9359)

题意：给定一个 $n \times m$ 网格。一些格子是障碍，其他为空，用 $k+1$ 种颜色 $0.1 \dots k$ 给空格子染色。不能有同一行或列的格子染相同颜色。给定  $c,d > 0 $，方案代价为 $ck+dz$。

对于每行和每列编号，在空格（下表中的 X）的横纵坐标之间连权值为 $1$ 的边。


建立虚拟源汇点，源点向行，列向汇点连权值为 $k$ 的边，跑从源点到汇点的最大流即为最多的非 $0$ 数。

如样例：
||4|5|6|7|
|-|-|-|-|-|
|1|X||||
|2||X|X||
|3|||X|X|

![graph](https://img.remit.ee/api/file/BQACAgUAAyEGAASHRsPbAAEF8UlpD0RLQoKcUoS1sMTlXwl37kao3QACIxoAAqr0gVQHHlLYkqShtjYE.png)

当行列直间的边的流量为 $1$ 时，即在该点填入了非 $0$ 的数。又源点到每个行点的边权为 $k$ 限制了该行可以填非 $0$ 数的个数小于等于 $k$，列也同理。


故这个方法可以在满足题目现在的情况下求出最多非零数。

又显然当 $k$ 为 $\max(n,m)$ 时可以给每个空格都填非 $0$ 数，可以枚举 $k$ 从 $0$ 到 $\max(n,m)$，求出最小代价。

复杂度 $O(n^4)$，可以通过。

代码：

```cpp
#include<bits/stdc++.h>
#ifndef ONLINE_JUDGE
// #define OPEN_FILE
// #define OPEN_TIME
#endif
#define AC return 0;
#define ll long long
#define ull unsigned long long
using namespace std;
const int maxn=1e6+10,INF=0x3f3f3f3f,mod=1e9+7;
const double eps=1e-8;
//cnt为空格数
ll n,m,c,d,s,t,cnt;
struct Edge{
    int v,w,nx;
}e[maxn];
int head[maxn],idx=1;//idx=1，使可以通过i^1来找到反向边
void AddEdge(int u,int v,int w){
    e[++idx]={v,w,head[u]};
    head[u]=idx;
}
int cur[maxn],dis[maxn];
bool Bfs(){//BFS分层
    queue<int>q;
    for(int i=1;i<=t;++i) 
        cur[i]=head[i],dis[i]=INF;
    q.push(s);
    dis[s]=0;
    while(!q.empty()){
        int u=q.front();q.pop();
        for(int i=head[u];i;i=e[i].nx){
            int v=e[i].v,w=e[i].w;
            if(dis[v]==INF&&w)
                dis[v]=dis[u]+1,q.push(v);
        }
    }
    return dis[t]!=INF;
}
int Dfs(int u,int las){
    if(u==t) return las;
    int res=0;
    for(int i=cur[u];i&&las;i=e[i].nx){
        cur[u]=i;//当前弧优化
        int v=e[i].v,w=min(e[i].w,las);
        if(dis[v]==dis[u]+1&&w){
            int k=Dfs(v,w);
            las-=k,res+=k;
            e[i].w-=k,e[i^1].w+=k;
        }
    }
    if(!res) dis[u]=INF;
    return res;
}
ll Flow(){//采用dinic跑最大流
    ll ans=0;
    while(Bfs())
        ans+=Dfs(s,INF);
    return ans;
}
void solve(){
    ll res=cnt*d;//当k为0时只能全部填0
    for(int k=1;k<=max(n,m);++k){
        //将图恢复为初始并改源汇点流量
        for(int i=head[s];i;i=e[i].nx) e[i].w=k,e[i^1].w=0;
        for(int i=head[t];i;i=e[i].nx) e[i^1].w=k,e[i].w=0;
        for(int u=1;u<=n;++u) for(int i=head[u];i;i=e[i].nx) if(e[i].v!=s)
            e[i].w=1,e[i^1].w=0;
        res=min(res,(cnt-Flow())*d+c*k);
    }
    cout<<res;
}
void init(){
    cin>>n>>m>>c>>d;
    //虚拟源汇点
    s=n+m+1,t=n+m+2;
    char ch;
    for(int i=1;i<=n;++i) 
        AddEdge(s,i,1),AddEdge(i,s,0);//最大流加双向边
    for(int i=n+1;i<=n+m;++i) 
        AddEdge(i,t,1),AddEdge(t,i,0);
    for(int u=1;u<=n;++u) for(int v=n+1;v<=n+m;++v){
        cin>>ch;
        if(ch=='.') 
            AddEdge(u,v,1),AddEdge(v,u,0),++cnt;
    }
}
int main(){
#ifdef OPEN_FILE
    freopen("in.txt","r",stdin);
    freopen("out.txt","w",stdout);
#endif
#ifdef OPEN_TIME
    auto StartTime=clock();
#endif
    // ios::sync_with_stdio(false),cin.tie(nullptr),cout.tie(nullptr);
    int T=1;
    // cin>>T;
    // while(cin>>n){
    while(T--){
        init();
        solve();
    }
#ifdef OPEN_TIME
    cerr<<"used: "<<(double)(clock()-StartTime)/CLOCKS_PER_SEC*1000<<" ms"<<endl;
#endif
    AC
}
```
