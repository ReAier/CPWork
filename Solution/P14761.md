## [P14761](https://www.luogu.com.cn/problem/P14761)

题意：一个括号序列，初始为空，每次插入一对括号并查询插入后有多少括号改变匹配。

观察样例可以发现每次改变匹配的是将插入区间内分离后的未匹配括号乘二，我们可以维护区间内的为匹配括号数。

我们可以用平衡树来维护支持插入的序列，节点信息为区间内未匹配的左或右括号数量，答案即为两种括号数相加。合并两个区间时需要减去区间之间的匹配括号数量。如：

```cpp
void PushUp(int p){
    siz(p)=siz(ls(p))+siz(rs(p))+1;
    int Pi=min(le(ls(p))+!val(p),ri(rs(p))+val(p));
    le(p)=le(ls(p))+le(rs(p))-Pi+!val(p);
    ri(p)=ri(ls(p))+ri(rs(p))-Pi+val(p);
}
```

其中 $le_p$ 为未匹配左括号数，$ri_p$ 为未匹配右括号数，$Pi$ 为区间间匹配数，$val_p$ 为当前节点信息，`0` 为 `(`，`1` 为 `)`。

我们使用 FHQTreap 实现平衡树，每次在 `l r` 间插入时分裂出 $[l+1,r]$ 区间，答案为区间为匹配左右括号数之和乘二，即 $le_p+ri_p$。然后合并时在分裂区间左右分别加上新建括号，如：

```cpp
int Insert(int L,int R){
    int l,p,r,ans;
    Split(rt,L,l,p);
    Split(p,R-L,p,r);
    ans=le(p)+ri(p);
    rt=Merge(Merge(l,NewNode(0)),Merge(p,Merge(NewNode(1),r)));
    return ans*2;
}
```

    struct Node{
        //le 未匹配左 ri 右

        bool val;//0 ( 1 )
    }t[maxn];
#define ls(p) t[p].ls

#define ri(p) t[p].ri

#define pri(p) t[p].pri
#define val(p) t[p].val
    int rt,idx;
    mt19937 mt;
    int NewNode(bool val){
        t[++idx]={0,0,!val,val,1,int(mt()),val};
        return idx;

    void PushUp(int p){
        siz(p)=siz(ls(p))+siz(rs(p))+1;
        int Pi=min(le(ls(p))+!val(p),ri(rs(p))+val(p));
        le(p)=le(ls(p))+le(rs(p))-Pi+!val(p);
        ri(p)=ri(ls(p))+ri(rs(p))-Pi+val(p);
    }
    int Merge(int l,int r){
        if(!l||!r) return l|r;
        if(pri(l)>pri(r)){
            rs(l)=Merge(rs(l),r);
            PushUp(l);
            return l;
        }else{
            ls(r)=Merge(l,ls(r));
            PushUp(r);
            return r;
        }
    }
    void Split(int p,int siz,int &L,int &R){
        if(!p) return L=R=0,void();
        if(siz(ls(p))+1<=siz){
            L=p;
            Split(rs(p),siz-siz(ls(p))-1,rs(p),R);
        }
        else{
            R=p;
            Split(ls(p),siz,L,ls(p));
        }
        PushUp(p);
    }
    int Insert(int L,int R){
        int l,p,r,ans;
        Split(rt,L,l,p);
        Split(p,R-L,p,r);
        ans=le(p)+ri(p);
        rt=Merge(Merge(l,NewNode(0)),Merge(p,Merge(NewNode(1),r)));
        return ans*2;
    }
